            if (sessions.length === 0) {
                tbody.innerHTML = '';
                empty.classList.remove('hidden');
                return;
            }
            
            empty.classList.add('hidden');
            tbody.innerHTML = '';
            
            sessions.forEach(session => {
                const tr = document.createElement('tr');
                tr.innerHTML = `
                    <td><span class="badge ${session.status === 'active' ? 'badge-success' : 'badge-warning'}">${session.status || 'completed'}</span></td>
                    <td><strong>${session.device_name || '-'}</strong></td>
                    <td>${session.user || 'admin'}</td>
                    <td>${formatDate(session.started_at)}</td>
                    <td>${formatDuration(session.duration_seconds)}</td>
                    <td>
                        <button class="action-btn" onclick="viewSession('${session.session_id}')" title="Visualizar">
                            <i class="fas fa-eye"></i>
                        </button>
                        <button class="action-btn" onclick="downloadSession('${session.session_id}')" title="Download">
                            <i class="fas fa-download"></i>
                        </button>
                    </td>
                `;
                tbody.appendChild(tr);
            });
        }

        function editDevice(id) {
            const device = devices.find(d => d.id === id);
            if (!device) return;

            document.getElementById('device-modal-title').textContent = 'Editar Dispositivo';
            document.getElementById('device-id').value = device.id;
            document.getElementById('device-name').value = device.hostname || '';
            document.getElementById('device-ip').value = device.ip_address || '';
            document.getElementById('device-type').value = device.is_bras ? 'bras' : (device.device_type || 'router');
            document.getElementById('device-vendor').value = device.vendor || '';
            document.getElementById('device-ssh-user').value = device.ssh_user || '';
            document.getElementById('device-ssh-password').value = device.ssh_password || '';
            document.getElementById('device-ssh-port').value = device.ssh_port || 22;
            document.getElementById('device-snmp-community').value = device.snmp_community || '';
            document.getElementById('device-snmp-port').value = device.snmp_port || 161;
            document.getElementById('device-backup-enabled').checked = device.backup_enabled !== false;

            document.getElementById('device-modal').classList.remove('hidden');
        }

        async function saveDevice() {
            const id = document.getElementById('device-id').value;
            const data = {
                id: id ? parseInt(id) : null,
                hostname: document.getElementById('device-name').value,
                ip_address: document.getElementById('device-ip').value,
                device_type: document.getElementById('device-type').value,
                is_bras: document.getElementById('device-type').value === 'bras',
                vendor: document.getElementById('device-vendor').value,
                ssh_user: document.getElementById('device-ssh-user').value,
                ssh_password: document.getElementById('device-ssh-password').value,
                ssh_port: parseInt(document.getElementById('device-ssh-port').value) || 22,
                snmp_community: document.getElementById('device-snmp-community').value,
                snmp_port: parseInt(document.getElementById('device-snmp-port').value) || 161,
                backup_enabled: document.getElementById('device-backup-enabled').checked,
            };

            try {
                const res = await fetch(`${API_BASE}/devices/save/`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(data)
                });
                const result = await res.json();
                if (result.status === 'success') {
                    closeDeviceModal();
                    loadDevices();
                    alert(id ? 'Dispositivo atualizado!' : 'Dispositivo criado!');
                } else {
                    alert('Erro: ' + (result.error || 'Erro desconhecido'));
                }
            } catch (err) {
                alert('Erro ao salvar dispositivo');
            }
        }

        // Delete Device
        function deleteDevice(id) {
            const device = devices.find(d => d.id === id);
            document.getElementById('delete-message').textContent = `Tem certeza que deseja excluir o dispositivo "${device?.hostname}"?`;

                await fetch(`${API_BASE}/devices/delete/`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ id })
                });
                loadDevices();
            };
            document.getElementById('delete-modal').classList.remove('hidden');
        }

        // Backup Functions
        async function viewBackup(id) {
            currentBackupId = id;
            const backup = backups.find(b => b.id === id);
            document.getElementById('backup-view-title').textContent = backup?.filename || 'Backup';
            
            try {
                const res = await fetch(`${API_BASE}/backups/download/?id=${id}`);
                const text = await res.text();
                document.getElementById('backup-content').textContent = text;
            } catch (err) {
                document.getElementById('backup-content').textContent = 'Erro ao carregar conteúdo';
            }
            
            document.getElementById('backup-view-modal').classList.remove('hidden');
        }

        function downloadBackup() {
            if (currentBackupId) {
                downloadBackupById(currentBackupId);
            }
        }

        function downloadBackupById(id) {
            window.open(`${API_BASE}/backups/download/?id=${id}`, '_blank');
        }

        function deleteBackup(id) {
            const backup = backups.find(b => b.id === id);
            document.getElementById('delete-message').textContent = `Tem certeza que deseja excluir o backup "${backup?.filename}"?`;
            deleteCallback = async () => {
                await fetch(`${API_BASE}/backups/delete/`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ id })
                });
                loadBackups();
            };
            document.getElementById('delete-modal').classList.remove('hidden');
        }

        // Run Backup
        async function runBackup(deviceId) {
            try {
                const res = await fetch(`${API_BASE}/devices/backup/run/`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ device_id: deviceId })
                });
                const result = await res.json();
                alert(result.status === 'success' ? 'Backup iniciado!' : 'Erro: ' + result.error);
            } catch (err) {
                alert('Erro ao executar backup');
            }
        }

        // Delete Modal
        function closeDeleteModal() {
            document.getElementById('delete-modal').classList.add('hidden');
            deleteCallback = null;
        }

        async function confirmDelete() {
            if (deleteCallback) {
                await deleteCallback();
            }
            closeDeleteModal();
